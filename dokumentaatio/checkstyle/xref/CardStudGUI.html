274     private void checkButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_checkButtonActionPerformed
275         // TODO add your handling code here:
276         if (card10.getText().isEmpty()) {
277             Deck deck = new Deck();
278             deck.shuffle();
279             ArrayList<String> usedCards = new ArrayList();
280             int i = 2;
281             for (JTextField field : cardFields) {
282                 if (!field.getText().isEmpty()) {
283                     usedCards.add(field.getText());
284                 } else {
285                     if (i > 0) {
286                         while (true) {
287                             Card card = deck.dealCard();
288                             if (!usedCards.contains(card.toString())) {
289                                 field.setText(card.toString());
290                                 break;
291                             }
292                         }                   
293                         i--;
294                     }
295                 }           
296             }
297         }
298         
299
