321     private void betButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_betButtonActionPerformed
322         try {
323             Thread.sleep(75);
324         } catch(InterruptedException ex) {
325             Thread.currentThread().interrupt();
326         }
327         // TODO add your handling code here:
328         int potSize = Integer.parseInt(potSizeField.getText());
329         int AIChips = Integer.parseInt(AIChipField.getText());
330         int playerChips = Integer.parseInt(playerChipField.getText());
331         int bet = Integer.parseInt(potSizeField.getText()) / 2;

333         AIChips -= bet;

341         } else {          
342             handleTheEndOfHand();
343         }

385     public void handleTheEndOfHand() {    
386             try {
387                 Thread.sleep(200);
388             } catch(InterruptedException ex) {
389                 Thread.currentThread().interrupt();
390             }
391             Card AICard1 = new Card(card2.getText());
392             Card AICard2 = new Card(card4.getText());
393             Card AICard3 = new Card(card6.getText());
394             Card AICard4 = new Card(card8.getText());
395             Card AICard5 = new Card(card10.getText());
396             Hand AIHand = new Hand(AICard1, AICard2, AICard3, AICard4, AICard5);
397             Card playerCard1 = new Card(card1.getText());
398             Card playerCard2 = new Card(card3.getText());
399             Card playerCard3 = new Card(card5.getText());
400             Card playerCard4 = new Card(card7.getText());
401             Card playerCard5 = new Card(card9.getText());
402             Hand playerHand = new Hand(playerCard1, playerCard2, playerCard3, playerCard4, playerCard5);    
403             messageField.setVisible(true);
404             if (playerHand.compareTo(AIHand) > 0) {
405                 int playerChips = Integer.parseInt(playerChipField.getText());
406                 playerChips += Integer.parseInt(potSizeField.getText());
407                 playerChipField.setText("" + playerChips);
408                 messageField.setText("Player wins");
409             } else if (playerHand.compareTo(AIHand) < 0) {
410                 int AIChips = Integer.parseInt(AIChipField.getText());
411                 AIChips += Integer.parseInt(potSizeField.getText());
412                 AIChipField.setText("" + AIChips);
413                 messageField.setText("AI wins");
414             } else {
415                 int playerChips = Integer.parseInt(playerChipField.getText());
416                 playerChips += Integer.parseInt(potSizeField.getText()) / 2;
417                 playerChipField.setText("" + playerChips);
418                 int AIChips = Integer.parseInt(AIChipField.getText());
419                 AIChips += Integer.parseInt(potSizeField.getText()) / 2;
420                 AIChipField.setText("" + AIChips);
421                 messageField.setText("Tie");
422             }
423             hiddenCard.setText(card2.getText());
424             if (hiddenCard.getText().contains("Hearts")) {
425                 hiddenCard.setForeground(Color.red);
426             } else if (hiddenCard.getText().contains("Diamonds")) {
427                 hiddenCard.setForeground(Color.blue);
428             } else if (hiddenCard.getText().contains("Clubs")) {
429                 hiddenCard.setForeground(Color.green);
430             } else {
431                 hiddenCard.setForeground(Color.BLACK);
432             }
433             hiddenCard.setBackground(Color.WHITE);
434             card1.setBackground(Color.WHITE);        
435             betButton.setText("Bet 1");            
436             drawButton.setVisible(true);
437             foldButton.setVisible(false);
438             checkButton.setVisible(false);
439             betButton.setVisible(false);
440     }

442     public void drawTwoCards() {
443         Deck deck = new Deck();
444             deck.shuffle();
445             ArrayList<String> usedCards = new ArrayList();
446             int i = 2;
447             for (JTextField field : cardFields) {
448                 if (!field.getText().isEmpty()) {
449                     usedCards.add(field.getText());
450                 } else {
451                     if (i > 0) {
452                         while (true) {
453                             Card card = deck.dealCard();
454                             if (!usedCards.contains(card.toString())) {
455                                 field.setText(card.toString());
456                                 break;
457                             }
458                         }                   
459                         i--;
460                     }
461                 }   
462                 if (field.getText().contains("Hearts")) {
463                     field.setForeground(Color.red);
464                 } else if (field.getText().contains("Diamonds")) {
465                     field.setForeground(Color.blue);
466                 } else if (field.getText().contains("Clubs")) {
467                     field.setForeground(Color.green);
468                 } else {
469                     field.setForeground(Color.BLACK);
470                 }
471                 
472             }
473     }
