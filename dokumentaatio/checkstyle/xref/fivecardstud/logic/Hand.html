17  public class Hand implements Comparable<Hand> {
18      
19      private ArrayList<Card> hand;
20      private Rank rank;

22      public Hand(Card c1, Card c2, Card c3, Card c4, Card c5) {
23          this.hand = new ArrayList();
24          hand.add(c1);
25          hand.add(c2);
26          hand.add(c3);
27          hand.add(c4);
28          hand.add(c5);
29          Collections.sort(hand);
30          rankHand();
31      }    
32      public void addCard(Card card) {
33          hand.add(card);
34      }
35      
36      public void rankHand() {
37          if (isStraightFlush()) {
38              int highestCard = 0;
39              if (hand.get(3).getValue() == 5 && hand.get(4).getValue() == 14) {
40                  highestCard = 5;
41              } else {
42                  highestCard = hand.get(4).getValue();
43              }
44              StraightFlush straightFlush = new StraightFlush(highestCard);
45              this.rank = straightFlush;
46          } else if (isFourOfAKind()) {        
47              FourOfAKind fourOfAKind = new FourOfAKind(hand.get(2).getValue());
48              this.rank = fourOfAKind;
49          } else if (isFullHouse()) {
50              FullHouse fullHouse = new FullHouse(hand.get(2).getValue());
51              this.rank = fullHouse;
52          } else if (isFlush()) {
53              Flush flush = new Flush(hand.get(4).getValue(), hand.get(3).getValue(), hand.get(2).getValue(), hand.get(1).getValue(), hand.get(0).getValue());
54              this.rank = flush;
55          } else if (isStraight()) {
56              int highestCard = 0;
57              if (hand.get(3).getValue() == 5 && hand.get(4).getValue() == 14) {
58                  highestCard = 5;
59              } else {
60                  highestCard = hand.get(4).getValue();
61              }
62              Straight straight = new Straight(highestCard);
63              this.rank = straight;
64          } else if (isThreeOfAKind()) {
65              ThreeOfAKind threeOfAKind = new ThreeOfAKind(hand.get(2).getValue());
66              this.rank = threeOfAKind;
67          } else if (isTwoPair()) {
68              int kicker = 0;
69              if (hand.get(0).getValue() != hand.get(1).getValue()) {
70                  kicker = hand.get(0).getValue();
71              } else if (hand.get(2).getValue() != hand.get(3).getValue()) {
72                  kicker = hand.get(2).getValue();
73              } else {
74                  kicker = hand.get(4).getValue();
75              }
76              TwoPair twoPair = new TwoPair(hand.get(3).getValue(), hand.get(1).getValue(), kicker);
77              this.rank = twoPair;
78          } else if (isPair()) {
79              int value = 0;
80              ArrayList<Integer> kickers = new ArrayList();
81              int pairLocation = 999;         
82              for (int i = 0; i < 4; i++) {
83                  if (hand.get(i).getValue() == hand.get(i + 1).getValue()) {
84                      value = hand.get(i).getValue();
85                      pairLocation = i;
86                  }
87              }
88              for (int i = 0; i < 5; i++) {
89                  if (i != pairLocation && i != pairLocation + 1) {
90                      kickers.add(hand.get(i).getValue());
91                  }
92              }
93              Pair pair = new Pair(value, kickers.get(2), kickers.get(1), kickers.get(0));
94              this.rank = pair;
95          } else {
96              HighCard highCard = new HighCard(hand.get(4).getValue(), hand.get(3).getValue(), hand.get(2).getValue(), hand.get(1).getValue(), hand.get(0).getValue());
97              this.rank = highCard;
98          }
99          
100     }

102     public boolean isStraightFlush() {
103         return isFlush() && isStraight();
104     }
105     
106     public boolean isFourOfAKind() {
107         Set<Integer> a = new TreeSet();
108         for (int i = 0; i < 4; i++) {
109             a.add(hand.get(i).getValue());
110         }
111         Set<Integer> b = new TreeSet();
112         for (int i = 1; i < 5; i++) {
113             b.add(hand.get(i).getValue());
114         }
115         if (a.size() == 1) {
116             return true;
117         }
118         return b.size() == 1;
119     }

121     public boolean isFullHouse() {
122         Set<Integer> a = new TreeSet();
123         for (int i = 0; i < 3; i++) {
124             a.add(hand.get(i).getValue());
125         }
126         Set<Integer> b = new TreeSet();
127         for (int i = 3; i < 5; i++) {
128             b.add(hand.get(i).getValue());
129         }
130         if (a.size() == 1 && b.size() == 1) {
131             return true;
132         }
133         Set<Integer> c = new TreeSet();
134         for (int i = 0; i < 2; i++) {
135             c.add(hand.get(i).getValue());
136         }
137         Set<Integer> d = new TreeSet();
138         for (int i = 2; i < 5; i++) {
139             d.add(hand.get(i).getValue());
140         }
141         return (c.size() == 1 && d.size() == 1);
142     }

144     public boolean isFlush() {
145         Set<Integer> a = new TreeSet();
146         for (int i = 0; i < 5; i++) {
147             a.add(hand.get(i).getSuit());
148         }
149         return a.size() == 1;
150     }

152     public boolean isStraight() {
153         if (hand.get(0).getValue() == hand.get(1).getValue() - 1) {
154             if (hand.get(1).getValue() == hand.get(2).getValue() - 1) {
155                 if (hand.get(2).getValue() == hand.get(3).getValue() - 1) {
156                     if (hand.get(3).getValue() == hand.get(4).getValue() - 1) {
157                         return true;
158                     }
159                 }
160             }
161         }
162         if (hand.get(0).getValue() == 2) {
163             if (hand.get(1).getValue() == 3) {
164                 if (hand.get(2).getValue() == 4) {
165                     if (hand.get(3).getValue() == 5) {
166                         if (hand.get(4).getValue() == 14) {
167                             return true;
168                         }
169                     }
170                 }
171             }
172         }
173         return false;
174     }

176     public boolean isThreeOfAKind() {
177         Set<Integer> a = new TreeSet();
178         for (int i = 0; i < 3; i++) {
179             a.add(hand.get(i).getValue());
180         }
181         Set<Integer> b = new TreeSet();
182         for (int i = 1; i < 4; i++) {
183             b.add(hand.get(i).getValue());
184         }
185         Set<Integer> c = new TreeSet();
186         for (int i = 2; i < 5; i++) {
187             c.add(hand.get(i).getValue());
188         }
189         return (a.size() == 1 || b.size() == 1 || c.size() == 1);
190     }
191     
192     public boolean isTwoPair() {
193         Set<Integer> a = new TreeSet();
194         for (int i = 0; i < 2; i++) {
195             a.add(hand.get(i).getValue());
196         }
197         Set<Integer> b = new TreeSet();
198         for (int i = 2; i < 5; i++) {
199             b.add(hand.get(i).getValue());
200         }
201         if (a.size() == 1 && b.size() == 2) {
202             return true;
203         }
204         
205         Set<Integer> c = new TreeSet();
206         for (int i = 0; i < 3; i++) {
207             c.add(hand.get(i).getValue());
208         }
209         Set<Integer> d = new TreeSet();
210         for (int i = 3; i < 5; i++) {
211             d.add(hand.get(i).getValue());
212         }
213         return c.size() == 2 && d.size() == 1;
214     }

216     public boolean isPair() {
217         Set<Integer> a = new TreeSet();
218         for (int i = 0; i < 5; i++) {
219             a.add(hand.get(i).getValue());
220         }
221         return a.size() == 4;
222     }

224     public Rank getRank() {
225         return rank;

228     @Override
229     public int compareTo(Hand h) {
230         if (rank.rankAsNumber() != h.getRank().rankAsNumber()) {
231             return rank.rankAsNumber() - h.getRank().rankAsNumber();
232         }
233         if (rank.rankAsNumber() == 9) {
234             StraightFlush ownSf = (StraightFlush) rank;
235             StraightFlush hSf = (StraightFlush) h.getRank();
236             return ownSf.getHighestCard() - hSf.getHighestCard();
237         }
238         if (rank.rankAsNumber() == 8) {
239             FourOfAKind ownF = (FourOfAKind) rank;
240             FourOfAKind hF = (FourOfAKind) h.getRank();
241             return ownF.getValue() - hF.getValue();
242         }
243         if (rank.rankAsNumber() == 7) {
244             FullHouse ownFh = (FullHouse) rank;
245             FullHouse hFh = (FullHouse) h.getRank();
246             return ownFh.getValueOfThreeOfAKind() - hFh.getValueOfThreeOfAKind();
247         }
248         if (rank.rankAsNumber() == 6) {
249             Flush ownF = (Flush) rank;
250             Flush hF = (Flush) h.getRank();
251             for (int i = 0; i < 5; i++) {
252                 if (ownF.getValues().get(i) != hF.getValues().get(i)) {
253                     return ownF.getValues().get(i) - hF.getValues().get(i);
254                 }
255             }
256             return 0;
257         }
258         if (rank.rankAsNumber() == 5) {
259             Straight ownS = (Straight) rank;
260             Straight hS = (Straight) h.getRank();
261             return ownS.getHighestCard() - hS.getHighestCard();
262         }
263         if (rank.rankAsNumber() == 4) {
264             ThreeOfAKind ownT = (ThreeOfAKind) rank;
265             ThreeOfAKind hT = (ThreeOfAKind) h.getRank();
266             return ownT.getValue() - hT.getValue();
267         }
268         if (rank.rankAsNumber() == 3) {
269             TwoPair ownTp = (TwoPair) rank;
270             TwoPair hTp = (TwoPair) h.getRank();
271             if (ownTp.getValueOfBiggerPair() != hTp.getValueOfBiggerPair()) {
272                 return ownTp.getValueOfBiggerPair() - hTp.getValueOfBiggerPair();
273             }    
274             if (ownTp.getValueOfSmallerPair() != hTp.getValueOfSmallerPair()) {
275                 return ownTp.getValueOfSmallerPair() - hTp.getValueOfSmallerPair();
276             }
277             return ownTp.getKicker() - hTp.getKicker();
278         }
279         if (rank.rankAsNumber() == 2) {
280             Pair ownP = (Pair) rank;
281             Pair hP = (Pair) h.getRank();
282             if (ownP.getValueOfPair() != hP.getValueOfPair()) {
283                 return ownP.getValueOfPair() - hP.getValueOfPair();
284             }
285             for (int i = 0; i < 3; i++) {
286                 if (ownP.getKickers().get(i) != hP.getKickers().get(i)) {
287                     return ownP.getKickers().get(i) - hP.getKickers().get(i);
288                 }
289             }
290             return 0;
291         }
292         HighCard ownHc = (HighCard) rank;
293         HighCard hHc = (HighCard) h.getRank();
294         for (int i = 0; i < 5; i++) {
295             if (ownHc.getValues().get(i) != hHc.getValues().get(i)) {
296                 return ownHc.getValues().get(i) - hHc.getValues().get(i);
297             }
298         }
299         return 0;
300     }
