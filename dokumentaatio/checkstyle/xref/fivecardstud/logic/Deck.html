6   /**
7    *  An object of type Deck represents a deck of playing cards.  The deck
8    *  is a regular poker deck that contains 52 regular cards and that can
9    *  also optionally include two Jokers.
10   */
11  public class Deck {
12  
13      private Card[] deck;
14      private int cardsUsed;
15  
16      public Deck() {
17          deck = new Card[52];
18          int i = 0; 
19          for (int suit = 1; suit < 5; suit++) {
20              for (int value = 2; value < 15; value++) {
21                  deck[i] = new Card(value, suit);
22                  i++;
23              }
24          }
25          cardsUsed = 0;
26      }
27  
28      public void shuffle() {
29          for (int i = deck.length - 1; i > 0; i--) {
30              Card temp = deck[i];
31              int random = (int) (Math.random() * (i + 1));
32              deck[i] = deck[random];
33              deck[random] = temp;
34          }
35          cardsUsed = 0;
36      }
37  
38      public int cardsLeft() {
39          return deck.length - cardsUsed;
40      }
41  
42      public Card dealCard() {
43          cardsUsed++;
44          return deck[cardsUsed - 1];
45      }
46  
47      public Card[] getDeck() {
48          return deck;
49      }
50      
51      
52  }
